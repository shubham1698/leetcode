class Solution {
    public int searchInsert(int[] nums, int target) {
       int i=0;
        for(i=0;i<=nums.length-1;i++)
        {
            if(nums[i]==target)
            {
               return i;
            }
            
        }
         for(i=0;i<=nums.length-1;i++)
        {
             if(nums[i]>target)
             {
                 break;
             }
         }
        return i;
    }
}
